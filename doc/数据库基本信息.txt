


CREATE TABLE `dictionary` (
  `id` int(5) NOT NULL AUTO_INCREMENT COMMENT '主键',
  `zdzwm` varchar(255) DEFAULT NULL COMMENT '字典中文名 字典名称',
  `zdywm` varchar(255) DEFAULT NULL COMMENT '字典英文名字典代码',
  `zxbz` varchar(1) NOT NULL DEFAULT '0' COMMENT '注销标志：0:否/1:是',
  `zdlx` varchar(1) NOT NULL DEFAULT '1' COMMENT '字典类型：1：下拉框，2：层级字典:',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=24 DEFAULT CHARSET=utf8 COMMENT='字典名表'



CREATE TABLE `dictionarylist` (
  `id` int(8) NOT NULL AUTO_INCREMENT COMMENT '主键',
  `zdid` int(3) NOT NULL COMMENT '字典名表主键（外键）',
  `zdzwm` varchar(225) NOT NULL COMMENT '字典中文名',
  `zdywm` varchar(225) NOT NULL COMMENT '字典英文名',
  `zxbz` varchar(1) NOT NULL DEFAULT '0' COMMENT '注销标志',
  `ctdm` varchar(8) DEFAULT NULL COMMENT '词条代码',
  `ctmc` varchar(225) NOT NULL COMMENT '词条名称',
  `ctlx` varchar(1) NOT NULL DEFAULT '0' COMMENT '词条类型[0：可修改；1：不可修改',
  PRIMARY KEY (`id`),
  KEY `referencekey_id` (`zdid`)
) ENGINE=InnoDB AUTO_INCREMENT=149 DEFAULT CHARSET=utf8 ROW_FORMAT=DYNAMIC COMMENT='字典项表'
